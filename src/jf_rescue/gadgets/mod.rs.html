22/// Variable to represent the state of the sponge.
23pub trait SpongeStateVar<T, F> {
24    /// The native field.
25    type Native;
26    /// Non-native field.
27    type NonNative;
28    /// How variable is represented in this Rescue(NonNative)StateVar.
29    type Var;
30}
31
32/// Trait for rescue circuit over native field.
33pub trait RescueGadget<R, T, F>
34where
35    R: SpongeStateVar<T, F>,
36{
37    /// Given an input state st_0 and an output state st_1, ensure that st_1 =
38    /// rescue_permutation(st_0)  where rescue_permutation is the instance
39    /// of the Rescue permutation defined by its respective constants
40    /// * `input_var` - variables corresponding to the input state
41    /// * `returns` - variables corresponding to the output state
42    fn rescue_permutation(&mut self, input_var: R) -> Result<R, CircuitError>;
43
44    /// Rescue based Pseudo Random Permutation (PRP)
45    /// * `key_var` - rescue state variable corresponding to the cipher key
46    /// * `input_var` - rescue state variable corresponding to the plaintext
47    /// * `returns` - state variable corresponding to the cipher text
48    fn prp(&mut self, key_var: &R, input_var: &R) -> Result<R, CircuitError>;
49
50    /// Sponge-based hashes from Rescue permutations
51    /// * `data_vars` - sponge input variables, `data_vars.len()` should be a
52    ///   positive integer that is a multiple of the sponge rate (i.e. 3)
53    /// * `num_output` - number of output variables
54    /// * `returns` - a vector of variables that refers to the sponge hash
55    ///   output
56    fn rescue_sponge_no_padding(
57        &mut self,
58        data_vars: &[R::Var],
59        num_output: usize,
60    ) -> Result<Vec<R::Var>, CircuitError>;
61
62    /// Sponge-based hashes from Rescue permutations
63    /// * `data_vars` - sponge input variables,
64    /// * `num_output` - number of output variables
65    /// * `returns` - a vector of variables that refers to the sponge hash
66    ///   output
67    fn rescue_sponge_with_padding(
68        &mut self,
69        data_vars: &[R::Var],
70        num_output: usize,
71    ) -> Result<Vec<R::Var>, CircuitError>;
72
73    /// Keyed sponge with rate=3 and a single output
74    /// * `key` - key variable
75    /// * `input` - input variables, length must be multiple of CRHF_RATE (3)
76    /// * `returns` a variable that refers to the output
77    fn rescue_keyed_sponge_no_padding(
78        &mut self,
79        key: R::Var,
80        data_vars: &[R::Var],
81    ) -> Result<R::Var, CircuitError>;
82
83    /// Similar to [`Self::rescue_keyed_sponge_no_padding`] except
84    /// `data_var` are padded with "zero_var" to make length multiple of
85    /// CRHF_RATE (3)
86    fn rescue_keyed_sponge_with_zero_padding(
87        &mut self,
88        key: R::Var,
89        data_vars: &[R::Var],
90    ) -> Result<R::Var, CircuitError>;
91
92    /// Return the round keys variables for the Rescue block cipher
93    /// * `mds_states` - Rescue MDS matrix
94    /// * `key_var` - state variable representing the cipher key
95    /// * `returns` - state variables corresponding to the scheduled keys
96    fn key_schedule(
97        &mut self,
98        mds_states: &RescueMatrix<R::Native>,
99        key_var: &R,
100        prp_instance: &PRP<R::Native>,
101    ) -> Result<Vec<R>, CircuitError>;
102
103    /// Create a variable representing a rescue state
104    /// * `state` - Rescue state
105    /// * `returns` - state variables corresponding to the state
106    fn create_rescue_state_variable(
107        &mut self,
108        state: &RescueVector<R::Native>,
109    ) -> Result<R, CircuitError>;
110
111    /// Return the variable corresponding to the output of the of the Rescue
112    /// PRP where the rounds keys have already been computed "dynamically"
113    /// * `input_var` - variable corresponding to the plain text
114    /// * `mds_states` - Rescue MDS matrix
115    /// * `key_vars` - variables corresponding to the scheduled keys
116    /// * `returns` -
117    fn prp_with_round_keys(
118        &mut self,
119        input_var: &R,
120        mds: &RescueMatrix<R::Native>,
121        keys_vars: &[R],
122    ) -> Result<R, CircuitError>;
123}

125pub(crate) trait PermutationGadget<R, T, F>: Circuit<F>
126where
127    R: SpongeStateVar<T, F>,
128    F: PrimeField,
129{
130    fn check_var_bound_rescue_state(&self, rescue_state: &R) -> Result<(), CircuitError>;
131
132    fn add_constant_state(
133        &mut self,
134        input_var: &R,
135        constant: &RescueVector<R::Native>,
136    ) -> Result<R, CircuitError>;
137
138    fn add_state(&mut self, left_state_var: &R, right_state_var: &R) -> Result<R, CircuitError>;
139
140    /// Given a state st_0=(x_1,...,x_w) and st_1=(y_1,...,y_w),
141    /// add the constraints that ensure we have y_i=x_i ^{1/5} for i in
142    /// [1,...,w]
143    /// * `input_var` - rescue state variables st_0
144    /// * `returns` - rescue state variables st_1
145    fn pow_alpha_inv_state(&mut self, input_var: &R) -> Result<R, CircuitError>;
146
147    /// Given an input state st_0 and an output state st_1, ensure that st_1 = M
148    /// st_0 + C where M is a Rescue matrix and c is a constant vector
149    /// * `input_var` - variables corresponding to the input state
150    /// * `matrix` - matrix M in the description above
151    /// * `constant` - constant c in the description above
152    /// * `returns` - variables corresponding to the output state
153    fn affine_transform(
154        &mut self,
155        input_var: &R,
156        matrix: &RescueMatrix<R::Native>,
157        constant: &RescueVector<R::Native>,
158    ) -> Result<R, CircuitError>;
159
160    /// Given an input state st_0=(x_1,...,x_w) and an output state
161    /// st_1=(y_1,...,y_m) y_i = \sum_{j=1}^w M_{i,j}x_j^alpha+c_i for all i in
162    /// [1,..,w] where M is a Rescue matrix and c=(c_1,...,c_w) is a
163    /// constant vector
164    /// * `input_var` - variables corresponding to the input state
165    /// * `matrix` - matrix M in the description above
166    /// * `constant` - constant c in the description above
167    /// * `returns` - variables corresponding to the output state
168    fn non_linear_transform(
169        &mut self,
170        input_var: &R,
171        matrix: &RescueMatrix<R::Native>,
172        constant: &RescueVector<R::Native>,
173    ) -> Result<R, CircuitError>;
174
175    /// Define a constraint such that y = x^(1/alpha).
176    /// It is implemented by setting q_{H1} y^alpha = q_O x
177    /// * `input_var`  - variable id corresponding to x in the equation above
178    /// * `returns` - the variable id corresponding to y
179    fn pow_alpha_inv(&mut self, input_var: R::Var) -> Result<R::Var, CircuitError>;
180
181    /// Given an input state st_0 and an output state st_1, ensure that st_1 is
182    /// obtained by applying the rescue permutation with a specific  list of
183    /// round keys (i.e. the keys are constants) and a matrix
184    /// * `input_var` - variables corresponding to the input state
185    /// * `mds` - Rescue matrix
186    /// * `round_keys` - list of round keys
187    /// * `returns` - variables corresponding to the output state
188    fn permutation_with_const_round_keys(
189        &mut self,
190        input_var: R,
191        mds: &RescueMatrix<R::Native>,
192        round_keys: &[RescueVector<R::Native>],
193    ) -> Result<R, CircuitError>;
194}
