7//! BLS Signature Scheme
8//!
9//! Conforming to [IRTF draft][irtf], wrapping [`blst` crate][blst] under the
10//! hood.
11//!
12//! [irtf]: https://datatracker.ietf.org/doc/pdf/draft-irtf-cfrg-bls-signature-05
13//! [blst]: https://github.com/supranational/blst
14//!
15//! # Examples
16//!
17//! ```
18//! use rand_chacha::{ChaCha20Rng, rand_core::SeedableRng};
19//! use jf_signature::{SignatureScheme, bls_over_bls12381::BLSSignatureScheme};
20//!
21//! let pp = BLSSignatureScheme::param_gen::<ChaCha20Rng>(None)?;
22//!
23//! // make sure the PRNG passed has good and trusted entropy.
24//! // you could use `OsRng` from `rand_core` or `getrandom` crate,
25//! // or a `SeedableRng` like `ChaChaRng` with seed generated from good randomness source.
26//! let mut rng = ChaCha20Rng::from_seed([0u8; 32]);
27//! let (sk, pk) = BLSSignatureScheme::key_gen(&pp, &mut rng)?;
28//!
29//! let msg = "The quick brown fox jumps over the lazy dog";
30//! let sig = BLSSignatureScheme::sign(&pp, &sk, &msg, &mut rng)?;
31//! assert!(BLSSignatureScheme::verify(&pp, &pk, &msg, &sig).is_ok());
32//!
33//! # Ok::<(), Box<dyn ark_std::error::Error>>(())
34//! ```
35//!
36//! ## Generating independent keys from the same IKM
37//!
38//! In case you want to keep the IKM for multiple key pairs, and potentially
39//! reconstruct them later on from IKM.
40//!
41//! ```
42//! use rand_chacha::{ChaCha20Rng, rand_core::{SeedableRng, RngCore}};
43//! use sha2::{Sha256, Digest};
44//! use jf_signature::{SignatureScheme, bls_over_bls12381::BLSSignatureScheme};
45//!
46//! let mut rng = ChaCha20Rng::from_seed([0u8; 32]); // seed from proper entropy source in practice!
47//! let pp = BLSSignatureScheme::param_gen::<ChaCha20Rng>(None)?;
48//!
49//! // NOTE: in practice, please use [`zeroize`][zeroize] to wipe sensitive
50//! // key materials out of memory.
51//! let mut ikm = [0u8; 32]; // should be at least 32 bytes
52//! rng.fill_bytes(&mut ikm);
53//!
54//! let mut hasher = Sha256::new();
55//! hasher.update(b"MY-BLS-SIG-KEYGEN-SALT-DOM-SEP");
56//! let salt = hasher.finalize();
57//!
58//! let (sk1, pk1) = BLSSignatureScheme::key_gen_v5(&ikm, &salt, b"banking".as_ref())?;
59//! let (sk2, pk2) = BLSSignatureScheme::key_gen_v5(&ikm, &salt, b"legal".as_ref())?;
60//!
61//! let msg = "I authorize transferring 10 dollars to Alice";
62//! let sig = BLSSignatureScheme::sign(&pp, &sk1, &msg, &mut rng)?;
63//! assert!(BLSSignatureScheme::verify(&pp, &pk1, &msg, &sig).is_ok());
64//!
65//! let msg = "I agree to the Terms and Conditions.";
66//! let sig = BLSSignatureScheme::sign(&pp, &sk2, &msg, &mut rng)?;
67//! assert!(BLSSignatureScheme::verify(&pp, &pk2, &msg, &sig).is_ok());
68//!
69//! # Ok::<(), Box<dyn ark_std::error::Error>>(())
70//! ```
71//!
72//! [zeroize]: https://github.com/RustCrypto/utils/tree/master/zeroize
73

97/// A BLS Secret Key (Signing Key). We intentionally omit the implementation of
98/// `serde::Serializable` so that it won't be unnoticably serialized or printed
99/// out through outer structs. However, users could manually serialize it into
100/// bytes by calling `to_bytes()` or `to_tagged_base64()` and exercise with
101/// self-cautions.
102pub struct BLSSignKey(SecretKey);
103
104// This content-hiding `Debug` implementation makes sure that the auto-derived
105// `Debug` for user's outer struct won't undesirably print out this secret key.
106impl core::fmt::Debug for BLSSignKey {
107    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
108        f.debug_struct("<BLSSignKey>").finish()
109    }
110}
111
112impl BLSSignKey {
113    /// Explicit calls to serialize a `SignKey` into bytes.
114    pub fn to_bytes(&self) -> [u8; 32] {
115        self.0.to_bytes()
116    }
117
118    /// Deserialize `SignKey` from bytes.
119    pub fn from_bytes(bytes: &[u8; 32]) -> Result<BLSSignKey, BLST_ERROR> {
120        SecretKey::from_bytes(bytes).map(|sk| BLSSignKey(sk))
121    }
122
123    /// Explicit calls to serialize a `SignKey` into `TaggedBase64`.
124    pub fn to_tagged_base64(&self) -> Result<TaggedBase64, Tb64Error> {
125        TaggedBase64::new(tag::BLS_SIGNING_KEY, &self.to_bytes())
126    }
127}

129impl<'a> TryFrom<&'a TaggedBase64> for BLSSignKey {
130    type Error = Tb64Error;
131
132    fn try_from(tb: &'a TaggedBase64) -> Result<Self, Self::Error> {
133        if tb.tag() != tag::BLS_SIGNING_KEY {
134            return Err(Tb64Error::InvalidTag);
135        }
136        SecretKey::from_bytes(tb.as_ref())
137            .map(|sk| BLSSignKey(sk))
138            .map_err(|err| Tb64Error::InvalidData)
139    }
140}

175impl CanonicalSerialize for BLSVerKey {
176    fn serialize_with_mode<W: Write>(
177        &self,
178        writer: W,
179        compress: Compress,
180    ) -> Result<(), SerializationError> {
181        if compress == Compress::No {
182            CanonicalSerialize::serialize_compressed(&self.serialize()[..], writer)
183        } else {
184            CanonicalSerialize::serialize_compressed(&self.compress()[..], writer)
185        }
186    }
187
188    fn serialized_size(&self, _compress: Compress) -> usize {
189        BLS_SIG_COMPRESSED_PK_SIZE
190    }
191
192    fn uncompressed_size(&self) -> usize {
193        BLS_SIG_PK_SIZE
194    }
195}

197impl CanonicalDeserialize for BLSVerKey {
198    fn deserialize_with_mode<R: Read>(
199        mut reader: R,
200        compress: Compress,
201        validate: Validate,
202    ) -> Result<Self, SerializationError> {
203        let len = <usize as ark_serialize::CanonicalDeserialize>::deserialize_with_mode(
204            &mut reader,
205            Compress::Yes,
206            validate,
207        )?;
208        // length validation is always performed
209        if !(len == BLS_SIG_COMPRESSED_PK_SIZE && compress == Compress::Yes
210            || len == BLS_SIG_PK_SIZE && compress == Compress::No)
211        {
212            return Err(SerializationError::InvalidData);
213        }
214
215        let pk = if compress == Compress::Yes {
216            let mut pk_bytes = [0u8; BLS_SIG_COMPRESSED_PK_SIZE];
217            reader.read_exact(&mut pk_bytes)?;
218
219            PublicKey::uncompress(&pk_bytes).map_err(|_| SerializationError::InvalidData)?
220        } else {
221            let mut pk_bytes = [0u8; BLS_SIG_PK_SIZE];
222            reader.read_exact(&mut pk_bytes)?;
223
224            PublicKey::deserialize(&pk_bytes).map_err(|_| SerializationError::InvalidData)?
225        };
226
227        let ver_key = Self(pk);
228        if validate == Validate::Yes && ver_key.check().is_err() {
229            return Err(SerializationError::InvalidData);
230        }
231
232        Ok(ver_key)
233    }
234}
235
236impl Valid for BLSVerKey {
237    fn check(&self) -> Result<(), ark_serialize::SerializationError> {
238        self.0
239            .validate()
240            .map_err(|_| SerializationError::InvalidData)
241    }
242}

249impl Deref for BLSSignature {
250    type Target = Signature;
251    fn deref(&self) -> &Self::Target {
252        &self.0
253    }
254}
255
256impl CanonicalSerialize for BLSSignature {
257    fn serialize_with_mode<W: Write>(
258        &self,
259        writer: W,
260        compress: Compress,
261    ) -> Result<(), SerializationError> {
262        if compress == Compress::No {
263            CanonicalSerialize::serialize_compressed(&self.serialize()[..], writer)
264        } else {
265            CanonicalSerialize::serialize_compressed(&self.compress()[..], writer)
266        }
267    }
268
269    fn serialized_size(&self, compress: Compress) -> usize {
270        if compress == Compress::Yes {
271            BLS_SIG_COMPRESSED_SIGNATURE_SIZE
272        } else {
273            BLS_SIG_SIGNATURE_SIZE
274        }
275    }
276}
277
278impl CanonicalDeserialize for BLSSignature {
279    // compressed
280    fn deserialize_with_mode<R: Read>(
281        mut reader: R,
282        compress: Compress,
283        validate: Validate,
284    ) -> Result<Self, SerializationError> {
285        let len = <usize as ark_serialize::CanonicalDeserialize>::deserialize_with_mode(
286            &mut reader,
287            Compress::Yes,
288            Validate::Yes,
289        )?;
290        if !(len == BLS_SIG_COMPRESSED_SIGNATURE_SIZE && compress == Compress::Yes
291            || len == BLS_SIG_SIGNATURE_SIZE && compress == Compress::No)
292        {
293            return Err(SerializationError::InvalidData);
294        }
295
296        let sig = if compress == Compress::Yes {
297            let mut sig_bytes = [0u8; BLS_SIG_COMPRESSED_SIGNATURE_SIZE];
298            reader.read_exact(&mut sig_bytes)?;
299            Signature::uncompress(&sig_bytes).map_err(|_| SerializationError::InvalidData)?
300        } else {
301            let mut sig_bytes = [0u8; BLS_SIG_SIGNATURE_SIZE];
302            reader.read_exact(&mut sig_bytes)?;
303
304            Signature::deserialize(&sig_bytes).map_err(|_| SerializationError::InvalidData)?
305        };
306
307        let bls_sig = Self(sig);
308
309        if validate == Validate::Yes && bls_sig.check().is_err() {
310            return Err(SerializationError::InvalidData);
311        }
312
313        Ok(bls_sig)
314    }
315}
316
317impl Valid for BLSSignature {
318    fn check(&self) -> Result<(), ark_serialize::SerializationError> {
319        self.0
320            .validate(true)
321            .map_err(|_| SerializationError::InvalidData)
322    }
323}

325/// BLS signature scheme. Wrapping around structs from the `blst` crate.
326/// See [module-level documentation](self) for example usage.
327#[derive(Clone, Debug)]
328pub struct BLSSignatureScheme;
329
330impl SignatureScheme for BLSSignatureScheme {
331    const CS_ID: &'static str = CS_ID_BLS_MIN_SIG;
332
333    /// Signing key
334    type SigningKey = BLSSignKey;
335
336    /// Verification key
337    type VerificationKey = BLSVerKey;
338
339    /// Public parameter
340    type PublicParameter = ();
341
342    /// Signature
343    type Signature = BLSSignature;
344
345    /// A message is &\[MessageUnit\]
346    type MessageUnit = u8;
347
348    /// generate public parameters from RNG.
349    /// If the RNG is not presented, use the default group generator.
350    fn param_gen<R: CryptoRng + RngCore>(
351        _prng: Option<&mut R>,
352    ) -> Result<Self::PublicParameter, SignatureError> {
353        Ok(())
354    }
355
356    /// Generate a BLS key pair.
357    /// Make sure the `prng` passed in are properly seeded with trusted entropy.
358    fn key_gen<R: CryptoRng + RngCore>(
359        _pp: &Self::PublicParameter,
360        prng: &mut R,
361    ) -> Result<(Self::SigningKey, Self::VerificationKey), SignatureError> {
362        let mut ikm = Zeroizing::new([0u8; 32]);
363        prng.fill_bytes(ikm.deref_mut());
364
365        let sk = SecretKey::key_gen(ikm.deref(), &[])?;
366        let vk = sk.sk_to_pk();
367        Ok((BLSSignKey(sk), BLSVerKey(vk)))
368    }
369
370    /// Sign a message
371    fn sign<R: CryptoRng + RngCore, M: AsRef<[Self::MessageUnit]>>(
372        _pp: &Self::PublicParameter,
373        sk: &Self::SigningKey,
374        msg: M,
375        _prng: &mut R,
376    ) -> Result<Self::Signature, SignatureError> {
377        Ok(BLSSignature(sk.0.sign(
378            msg.as_ref(),
379            Self::CS_ID.as_bytes(),
380            &[],
381        )))
382    }
383
384    /// Verify a signature.
385    fn verify<M: AsRef<[Self::MessageUnit]>>(
386        _pp: &Self::PublicParameter,
387        vk: &Self::VerificationKey,
388        msg: M,
389        sig: &Self::Signature,
390    ) -> Result<(), SignatureError> {
391        vk.check()
392            .map_err(|_| SignatureError::FailedValidityCheck)?;
393        sig.check()
394            .map_err(|_| SignatureError::FailedValidityCheck)?;
395        match sig.verify(false, msg.as_ref(), Self::CS_ID.as_bytes(), &[], vk, true) {
396            BLST_ERROR::BLST_SUCCESS => Ok(()),
397            e => Err(SignatureError::VerificationError(format!("{e:?}"))),
398        }
399    }
400}
401
402impl BLSSignatureScheme {
403    /// Alternative deterministic key_gen compatible with [IRTF draft v5][v5].
404    ///
405    /// - Secret byte string `ikm` MUST be infeasible to guess, ideally
406    ///   generated by a trusted source of randomness. `ikm` MUST be at least 32
407    ///   bytes long, but it MAY be longer.
408    /// - `salt` should either be empty or an unstructured byte string. It is
409    ///   RECOMMENDED to fix a uniformly random byte string of length 32. See
410    ///   details [here][salt].
411    /// - `key_info` is optional, it MAY be used to derived multiple independent
412    ///   keys from the same `ikm`. By default, `key_info` is the empty string.
413    ///
414    /// [v5]: https://datatracker.ietf.org/doc/pdf/draft-irtf-cfrg-bls-signature-05
415    /// [salt]: https://www.ietf.org/archive/id/draft-irtf-cfrg-bls-signature-05.html#name-choosing-a-salt-value-for-k
416    pub fn key_gen_v5(
417        ikm: &[u8],
418        salt: &[u8],
419        key_info: &[u8],
420    ) -> Result<
421        (
422            <Self as SignatureScheme>::SigningKey,
423            <Self as SignatureScheme>::VerificationKey,
424        ),
425        SignatureError,
426    > {
427        let sk = SecretKey::key_gen_v5(ikm, salt, key_info)?;
428        let vk = sk.sk_to_pk();
429
430        Ok((BLSSignKey(sk), BLSVerKey(vk)))
431    }
432}

448    #[test]
449    fn test_canonical_serde() {
450        let mut rng = jf_utils::test_rng();
451        let (sk, pk) = BLSSignatureScheme::key_gen(&(), &mut rng).unwrap();
452        let msg = "The quick brown fox jumps over the lazy dog";
453        let sig = BLSSignatureScheme::sign(&(), &sk, msg, &mut rng).unwrap();
454
455        let bytes = sk.to_bytes();
456        let de = BLSSignKey::from_bytes(&bytes).unwrap();
457        assert_eq!(sk, de);
458
459        let tagged_blob = sk.to_tagged_base64().unwrap();
460        let de: BLSSignKey = tagged_blob.try_into().unwrap();
461        assert_eq!(sk, de);
462
463        test_canonical_serde_helper(pk);
464        test_canonical_serde_helper(sig);
465    }
466
467    fn test_canonical_serde_helper<T>(data: T)
468    where
469        T: CanonicalSerialize + CanonicalDeserialize + Debug + PartialEq,
470    {
471        let mut bytes = vec![];
472        CanonicalSerialize::serialize_compressed(&data, &mut bytes).unwrap();
473        let de: T = CanonicalDeserialize::deserialize_compressed(&bytes[..]).unwrap();
474        assert_eq!(data, de);
475
476        bytes = vec![];
477        CanonicalSerialize::serialize_uncompressed(&data, &mut bytes).unwrap();
478        let de: T = CanonicalDeserialize::deserialize_uncompressed(&bytes[..]).unwrap();
479        assert_eq!(data, de);
480
481        bytes = vec![];
482        CanonicalSerialize::serialize_compressed(&data, &mut bytes).unwrap();
483        let de: T = CanonicalDeserialize::deserialize_compressed_unchecked(&bytes[..]).unwrap();
484        assert_eq!(data, de);
485    }
