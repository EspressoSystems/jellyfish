41/// Signature error type
42#[derive(Debug, Display, Eq, PartialEq)]
43pub enum SignatureError {
44    /// Bad parameter in function call, {0}
45    ParameterError(String),
46    /// Value is not in the right subgroup
47    FailedSubgroupCheck,
48    /// Value is not on the right elliptic curve
49    FailedOnCurveCheck,
50    // union over `FailedSubgroupCheck` and `FailedOnCurveCheck`
51    /// Value is not valid (possible cause: not on curve, in wrong subgroup,)
52    FailedValidityCheck,
53    /// Verification failed, {0}
54    VerificationError(String),
55}
56
57impl ark_std::error::Error for SignatureError {}
58
59impl From<BLST_ERROR> for SignatureError {
60    fn from(e: BLST_ERROR) -> Self {
61        match e {
62            BLST_ERROR::BLST_SUCCESS => {
63                Self::ParameterError("Expecting an error, but got a success.".to_string())
64            },
65            BLST_ERROR::BLST_VERIFY_FAIL => Self::VerificationError(format!("{e:?}")),
66            _ => Self::ParameterError(format!("{e:?}")),
67        }
68    }
69}

71/// Trait definition for a signature scheme.
72// A signature scheme is associated with a hash function H that is
73// to be used for challenge generation.
74// FIXME: update H bound once hash-api is merged.
75pub trait SignatureScheme: Clone + Send + Sync + 'static {
76    /// Ciphersuite Identifier
77    const CS_ID: &'static str;
78
79    /// Signing key.
80    type SigningKey: Debug + Clone + Send + Sync + Zeroize + PartialEq + Eq;
81
82    /// Verification key
83    type VerificationKey: Debug
84        + Clone
85        + Send
86        + Sync
87        + for<'a> Deserialize<'a>
88        + Serialize
89        + PartialEq
90        + Eq;
91
92    /// Public Parameter
93    type PublicParameter: Debug + Clone + PartialEq + for<'a> Deserialize<'a> + Serialize;
94
95    /// Signature
96    type Signature: Debug
97        + Clone
98        + Send
99        + Sync
100        + for<'a> Deserialize<'a>
101        + Serialize
102        + PartialEq
103        + Eq;
104
105    /// A message is &\[MessageUnit\]
106    type MessageUnit: Debug + Clone + Send + Sync;
107
108    /// generate public parameters from RNG.
109    /// If the RNG is not presented, use the default group generator.
110    // FIXME: the API looks a bit strange when the default generator is used.
111    // For example:
112    //   `S::param_gen::<StdRng>(None)`
113    // where `StdRng` is redundant.
114    fn param_gen<R: CryptoRng + RngCore>(
115        prng: Option<&mut R>,
116    ) -> Result<Self::PublicParameter, SignatureError>;
117
118    /// Sample a pair of keys.
119    fn key_gen<R: CryptoRng + RngCore>(
120        pp: &Self::PublicParameter,
121        prng: &mut R,
122    ) -> Result<(Self::SigningKey, Self::VerificationKey), SignatureError>;
123
124    /// Sign a message with the signing key
125    fn sign<R: CryptoRng + RngCore, M: AsRef<[Self::MessageUnit]>>(
126        pp: &Self::PublicParameter,
127        sk: &Self::SigningKey,
128        msg: M,
129        prng: &mut R,
130    ) -> Result<Self::Signature, SignatureError>;
131
132    /// Verify a signature.
133    fn verify<M: AsRef<[Self::MessageUnit]>>(
134        pp: &Self::PublicParameter,
135        vk: &Self::VerificationKey,
136        msg: M,
137        sig: &Self::Signature,
138    ) -> Result<(), SignatureError>;
139}

141/// Trait for aggregatable signatures.
142/// TODO: generic over hash functions
143// NOTE: we +Debug here instead of on `SignatureSchemes` because `schnorr <P:
144// CurveConfig>` doesn't impl Debug
145pub trait AggregateableSignatureSchemes:
146    SignatureScheme + Serialize + for<'a> Deserialize<'a> + Debug
147{
148    /// Aggregate multiple signatures into a single signature
149    /// The list of public keys is also in the input as some aggregate signature
150    /// schemes might also use pks for aggregation
151    fn aggregate(
152        pp: &Self::PublicParameter,
153        vks: &[Self::VerificationKey],
154        sigs: &[Self::Signature],
155    ) -> Result<Self::Signature, SignatureError>;
156
157    /// Verify an aggregate signature w.r.t. a list of messages and public keys.
158    /// It is user's responsibility to ensure that the public keys are
159    /// validated.
160    fn aggregate_verify<M: AsRef<[Self::MessageUnit]>>(
161        pp: &Self::PublicParameter,
162        vks: &[Self::VerificationKey],
163        msgs: &[M],
164        sig: &Self::Signature,
165    ) -> Result<(), SignatureError>;
166
167    /// Verify a multisignature w.r.t. a single message and a list of public
168    /// keys. It is user's responsibility to ensure that the public keys are
169    /// validated.
170    fn multi_sig_verify(
171        pp: &Self::PublicParameter,
172        vks: &[Self::VerificationKey],
173        msg: &[Self::MessageUnit],
174        sig: &Self::Signature,
175    ) -> Result<(), SignatureError>;
176}

178#[cfg(test)]
179mod tests {
180    use super::*;
181    use ark_std::{rand::prelude::StdRng, vec, vec::Vec};

184    pub(crate) fn sign_and_verify<S: SignatureScheme>(message: &[S::MessageUnit]) {
185        let rng = &mut test_rng();
186        let parameters = S::param_gen(Some(rng)).unwrap();
187        let (sk, pk) = S::key_gen(&parameters, rng).unwrap();
188        let sig = S::sign(&parameters, &sk, message, rng).unwrap();
189        assert!(S::verify(&parameters, &pk, message, &sig).is_ok());
190
191        let parameters = S::param_gen::<StdRng>(None).unwrap();
192        let (sk, pk) = S::key_gen(&parameters, rng).unwrap();
193        let sig = S::sign(&parameters, &sk, message, rng).unwrap();
194        assert!(S::verify(&parameters, &pk, message, &sig).is_ok());
195    }
196
197    pub(crate) fn agg_sign_and_verify<S: AggregateableSignatureSchemes>(
198        messages: &[&[S::MessageUnit]],
199        bad_message: &[S::MessageUnit],
200    ) {
201        let rng = &mut test_rng();
202        let parameters = S::param_gen(Some(rng)).unwrap();
203        let mut pks = vec![];
204        let mut sigs = vec![];
205        let mut partial_sigs = vec![];
206        let message_for_msig = messages[0];
207        for message in messages.iter() {
208            let (sk, pk) = S::key_gen(&parameters, rng).unwrap();
209            let sig = S::sign(&parameters, &sk, message, rng).unwrap();
210            let partial_sig = S::sign(&parameters, &sk, message_for_msig, rng).unwrap();
211            pks.push(pk);
212            sigs.push(sig);
213            partial_sigs.push(partial_sig);
214        }
215        // happy paths
216        let agg_sig = S::aggregate(&parameters, &pks, &sigs).unwrap();
217        let multi_sig = S::aggregate(&parameters, &pks, &partial_sigs).unwrap();
218        assert!(S::aggregate_verify(&parameters, &pks, messages, &agg_sig).is_ok());
219        assert!(S::multi_sig_verify(&parameters, &pks, message_for_msig, &multi_sig).is_ok());
220        // wrong messages length
221        assert!(S::aggregate_verify(&parameters, &pks, &messages[1..], &agg_sig).is_err());
222        // empty pks
223        assert!(S::aggregate_verify(&parameters, &[], messages, &agg_sig).is_err());
224        assert!(S::multi_sig_verify(&parameters, &[], message_for_msig, &multi_sig).is_err());
225        // wrong message
226        let mut bad_messages: Vec<&[S::MessageUnit]> = messages.to_vec();
227        bad_messages[0] = bad_message;
228        assert!(S::aggregate_verify(&parameters, &pks, &bad_messages, &agg_sig).is_err());
229        assert!(S::multi_sig_verify(&parameters, &pks, bad_message, &multi_sig).is_err());
230    }

232    pub(crate) fn failed_verification<S: SignatureScheme>(
233        message: &[S::MessageUnit],
234        bad_message: &[S::MessageUnit],
235    ) {
236        let rng = &mut test_rng();

238        let (sk, pk) = S::key_gen(&parameters, rng).unwrap();

241    }
