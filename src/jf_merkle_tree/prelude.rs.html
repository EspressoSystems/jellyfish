33/// Wrapper for rescue hash function
34#[derive(Clone, Copy, Debug, PartialEq, Eq)]
35pub struct RescueHash<F: RescueParameter> {
36    phantom_f: PhantomData<F>,
37}
38
39/// domain separator of algebraic hash, for the leaf node
40fn leaf_hash_dom_sep<F: PrimeField>() -> F {
41    F::one()
42}
43
44/// domain separator of algebraic hash, for the internal node
45fn internal_hash_dom_sep<F: PrimeField>() -> F {
46    F::zero()
47}
48
49/// domain separator of byte-oriented hash, for the leaf node
50pub(crate) const LEAF_HASH_DOM_SEP: &'static [u8; 1] = b"1";
51/// domain separator of byte-oriented hash, for the internal node
52pub(crate) const INTERNAL_HASH_DOM_SEP: &'static [u8; 1] = b"1";
53
54impl<I: Index, F: RescueParameter + From<I>> DigestAlgorithm<F, I, F> for RescueHash<F> {
55    fn digest(data: &[F]) -> Result<F, MerkleTreeError> {
56        let mut input = vec![internal_hash_dom_sep()];
57        input.extend(data.iter());
58        Ok(RescueCRHF::<F>::sponge_with_zero_padding(&input, 1)[0])
59    }
60
61    fn digest_leaf(pos: &I, elem: &F) -> Result<F, MerkleTreeError> {
62        let data = [leaf_hash_dom_sep(), F::from(pos.clone()), *elem];
63        Ok(RescueCRHF::<F>::sponge_with_zero_padding(&data, 1)[0])
64    }
65}

76// Make `FixedLenPoseidon2Hash<F, S, INPUT_SIZE, 1>` usable as Merkle tree hash
77// for arity INPUT_SIZE - 1. The first input element is used for the domain
78// separation.
79impl<I, F, S, const INPUT_SIZE: usize> DigestAlgorithm<F, I, F>
80    for FixedLenPoseidon2Hash<F, S, INPUT_SIZE, 1>
81where
82    I: Index,
83    F: PrimeField + From<I> + nimue::Unit,
84    S: Sponge<U = F>,
85{
86    fn digest(data: &[F]) -> Result<F, MerkleTreeError> {
87        let mut input = vec![internal_hash_dom_sep()];
88        input.extend(data.iter());
89        Ok(FixedLenPoseidon2Hash::<F, S, INPUT_SIZE, 1>::evaluate(input)?[0])
90    }
91
92    fn digest_leaf(pos: &I, elem: &F) -> Result<F, MerkleTreeError> {
93        let mut input = vec![leaf_hash_dom_sep(), F::from(pos.clone()), *elem];
94        Ok(FixedLenPoseidon2Hash::<F, S, INPUT_SIZE, 1>::evaluate(input)?[0])
95    }
96}

98/// Implement Internal node type and implement DigestAlgorithm for a hash
99/// function with 32 bytes output size
100///
101/// # Usage
102/// `impl_mt_hash_256!(Sha3_256, Sha3Node, Sha3Digest)` will
103/// - introduce `struct Sha3Node` for internal node in a merkle tree that use
104///   Sha3_256 as hash
105/// - introduce `struct Sha3Digest` which implements `DigestAlgorithm`
106macro_rules! impl_mt_hash_256 {
107    ($hasher:ident, $node_name:ident, $digest_name:ident) => {
108        /// Internal node for merkle tree
109        #[derive(Default, Eq, PartialEq, Clone, Copy, Ord, PartialOrd, Hash)]
110        pub struct $node_name(pub(crate) [u8; 32]);
111
112        impl fmt::Debug for $node_name {
113            fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
114                f.debug_tuple(&stringify!($node_name))
115                    .field(&hex::encode(self.0))
116                    .finish()
117            }
118        }
119
120        impl AsRef<[u8]> for $node_name {
121            fn as_ref(&self) -> &[u8] {
122                &self.0
123            }
124        }
125        impl CanonicalSerialize for $node_name {
126            fn serialize_with_mode<W: Write>(
127                &self,
128                mut writer: W,
129                _compress: Compress,
130            ) -> Result<(), SerializationError> {
131                writer.write_all(&self.0)?;
132                Ok(())
133            }
134            fn serialized_size(&self, _compress: Compress) -> usize {
135                32
136            }
137        }
138        impl CanonicalDeserialize for $node_name {
139            fn deserialize_with_mode<R: Read>(
140                mut reader: R,
141                _compress: Compress,
142                _validate: Validate,
143            ) -> Result<Self, SerializationError> {
144                let mut ret = [0u8; 32];
145                reader.read_exact(&mut ret)?;
146                Ok(Self(ret))
147            }
148        }
149        impl Valid for $node_name {
150            fn check(&self) -> Result<(), SerializationError> {
151                Ok(())
152            }
153        }
154
155        /// Wrapper for the actual hash function
156        #[derive(Clone, Debug, Hash, Eq, PartialEq)]
157        pub struct $digest_name;
158        impl<E: Element + CanonicalSerialize, I: Index> DigestAlgorithm<E, I, $node_name>
159            for $digest_name
160        {
161            fn digest(data: &[$node_name]) -> Result<$node_name, MerkleTreeError> {
162                let mut h = $hasher::new();
163                h.update(INTERNAL_HASH_DOM_SEP);
164                for value in data {
165                    h.update(value);
166                }
167                Ok($node_name(h.finalize().into()))
168            }
169
170            fn digest_leaf(_pos: &I, elem: &E) -> Result<$node_name, MerkleTreeError> {
171                let mut writer = Vec::new();
172                elem.serialize_compressed(&mut writer).unwrap();
173                let mut h = $hasher::new();
174                h.update(LEAF_HASH_DOM_SEP);
175                h.update(writer);
176                Ok($node_name(h.finalize().into()))
177            }
178        }
179    };
180}

196#[cfg(test)]
197mod tests {
198    use super::{MerkleTreeScheme, RescueMerkleTree};
199    use ark_bls12_377::Fr as Fr377;
200    use ark_bls12_381::Fr as Fr381;
201    use ark_bn254::Fr as Fr254;
202    use jf_rescue::{crhf::RescueCRHF, RescueParameter};
203
204    #[test]
205    fn test_extension_attack() {
206        test_extension_attack_helper::<Fr254>();
207        test_extension_attack_helper::<Fr377>();
208        test_extension_attack_helper::<Fr381>();
209    }
210
211    fn test_extension_attack_helper<F: RescueParameter>() {
212        let forged_val = F::from(42u64);
213        let attack_pos = 5u64;
214        let forged_pos = attack_pos * 3 + 2;
215        let data = [F::zero(), F::from(forged_pos), forged_val];
216        let val = RescueCRHF::<F>::sponge_no_padding(&data, 1).unwrap()[0];
217        let elems = ark_std::vec![val; attack_pos as usize + 1];
218        let mt = RescueMerkleTree::<F>::from_elems(None, elems).unwrap();
219        let commit = mt.commitment();
220        let (elem, mut proof) = mt.lookup(attack_pos).expect_ok().unwrap();
221        assert!(
222            RescueMerkleTree::<F>::verify(&commit, attack_pos, elem, &proof)
223                .unwrap()
224                .is_ok()
225        );
226        proof
227            .0
228            .insert(0, ark_std::vec![F::zero(), F::from(attack_pos)]);
229        assert!(
230            RescueMerkleTree::<F>::verify(&commit, forged_pos, forged_val, &proof)
231                .unwrap()
232                .is_err()
233        );
234    }
