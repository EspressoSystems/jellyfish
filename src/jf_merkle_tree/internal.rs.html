18/// An internal Merkle node.
19#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
20#[serde(bound = "E: CanonicalSerialize + CanonicalDeserialize,
21                 I: CanonicalSerialize + CanonicalDeserialize,")]
22pub enum MerkleNode<E: Element, I: Index, T: NodeValue> {
23    /// An empty subtree.
24    Empty,
25    /// An internal branching node
26    Branch {
27        /// Merkle hash value of this subtree
28        #[serde(with = "canonical")]
29        value: T,
30        /// All it's children
31        children: Vec<Arc<MerkleNode<E, I, T>>>,
32    },
33    /// A leaf node
34    Leaf {
35        /// Merkle hash value of this leaf
36        #[serde(with = "canonical")]
37        value: T,
38        /// Index of this leaf
39        #[serde(with = "canonical")]
40        pos: I,
41        /// Associated element of this leaf
42        #[serde(with = "canonical")]
43        elem: E,
44    },
45    /// The subtree is forgotten from the memory
46    ForgottenSubtree {
47        /// Merkle hash value of this forgotten subtree
48        #[serde(with = "canonical")]
49        value: T,
50    },
51}
52
53impl<E, I, T> MerkleNode<E, I, T>
54where
55    E: Element,
56    I: Index,
57    T: NodeValue,
58{
59    /// Return the value of this [`MerkleNode`].
60    #[inline]
61    pub(crate) fn value(&self) -> T {
62        match self {
63            Self::Empty => T::default(),
64            Self::Leaf {
65                value,
66                pos: _,
67                elem: _,
68            } => *value,
69            Self::Branch { value, children: _ } => *value,
70            Self::ForgottenSubtree { value } => *value,
71        }
72    }
73
74    #[inline]
75    pub(crate) fn is_forgotten(&self) -> bool {
76        matches!(self, Self::ForgottenSubtree { .. })
77    }
78}

80/// A (non)membership Merkle proof consists of all values of siblings of a

85#[tagged("MERKLE_PROOF")]
86pub struct MerkleTreeProof<T: NodeValue>(pub Vec<Vec<T>>);

88impl<T: NodeValue> super::MerkleProof<T> for MerkleTreeProof<T> {
89    /// Expected height of the Merkle tree.
90    fn height(&self) -> usize {
91        self.0.len()
92    }
93
94    /// Return all values of siblings of this Merkle path
95    fn path_values(&self) -> &[Vec<T>] {
96        &self.0
97    }

100/// Verify a merkle proof
101/// * `commitment` - a merkle tree commitment
102/// * `pos` - zero-based index of the leaf in the tree
103/// * `element` - the leaf value, None if verifying a non-membership proof
104/// * `proof` - a membership proof for `element` at given `pos`
105/// * `returns` - Ok(true) if the proof is accepted, Ok(false) if not. Err() if
106///   the proof is not well structured, E.g. not for this merkle tree.
107pub(crate) fn verify_merkle_proof<E, H, I, const ARITY: usize, T>(
108    commitment: &T,
109    pos: &I,
110    element: Option<&E>,
111    proof: &[Vec<T>],
112) -> Result<VerificationResult, MerkleTreeError>
113where
114    E: Element,
115    I: Index + ToTraversalPath<ARITY>,
116    T: NodeValue,
117    H: DigestAlgorithm<E, I, T>,
118{
119    let init = if let Some(elem) = element {
120        H::digest_leaf(pos, elem)?
121    } else {
122        T::default()
123    };
124    match element {
125        // only strictly checking this during membership proof
126        // in non-membership proof, empty leaf/branch node can have empty vector in
127        // their merkle proof
128        Some(_) => {
129            if proof.iter().any(|v| v.len() != ARITY - 1) {
130                return Err(MerkleTreeError::InconsistentStructureError(
131                    "Malformed proof".to_string(),
132                ));
133            }
134        },
135        None => {
136            if !proof.iter().all(|v| v.len() == ARITY - 1 || v.is_empty()) {
137                return Err(MerkleTreeError::InconsistentStructureError(
138                    "Malformed proof".to_string(),
139                ));
140            }
141        },
142    };
143    if element.is_some() && proof.iter().any(|v| v.len() != ARITY - 1) {
144        // only strictly checking this during membership proof
145        // in non-membership proof, empty leaf/branch node can have empty vector in
146        // their merkle proof
147        return Err(MerkleTreeError::InconsistentStructureError(
148            "Malformed proof".to_string(),
149        ));
150    }
151    let mut data = [T::default(); ARITY];
152    let computed_root = pos
153        .to_traversal_path(proof.len())
154        .iter()
155        .zip(proof.iter())
156        .try_fold(
157            init,
158            |val, (branch, values)| -> Result<T, MerkleTreeError> {
159                if values.is_empty() {
160                    Ok(T::default())
161                } else {
162                    data[..*branch].copy_from_slice(&values[..*branch]);
163                    data[*branch] = val;
164                    data[*branch + 1..].copy_from_slice(&values[*branch..]);
165                    H::digest(&data)
166                }
167            },
168        )?;
169    if computed_root == *commitment {
170        Ok(SUCCESS)
171    } else {
172        Ok(FAIL)
173    }
174}

176#[allow(clippy::type_complexity)]
177pub(crate) fn build_tree_internal<E, H, const ARITY: usize, T>(
178    height: Option<usize>,
179    elems: impl IntoIterator<Item = impl Borrow<E>>,
180) -> Result<(Arc<MerkleNode<E, u64, T>>, usize, u64), MerkleTreeError>
181where
182    E: Element,
183    H: DigestAlgorithm<E, u64, T>,
184    T: NodeValue,
185{
186    let leaves: Vec<_> = elems.into_iter().collect();
187    let num_leaves = leaves.len() as u64;
188    let height = height.unwrap_or_else(|| {
189        let mut height = 0usize;
190        let mut capacity = 1;
191        while capacity < num_leaves {
192            height += 1;
193            capacity *= ARITY as u64;
194        }
195        height
196    });
197    let capacity = BigUint::from(ARITY as u64).pow(height as u32);
198
199    if BigUint::from(num_leaves) > capacity {
200        Err(MerkleTreeError::ExceedCapacity)
201    } else if num_leaves == 0 {
202        Ok((Arc::new(MerkleNode::<E, u64, T>::Empty), height, 0))
203    } else if height == 0usize {
204        Ok((
205            Arc::new(MerkleNode::Leaf {
206                value: H::digest_leaf(&0, leaves[0].borrow())?,
207                pos: 0,
208                elem: leaves[0].borrow().clone(),
209            }),
210            height,
211            1,
212        ))
213    } else {
214        let mut cur_nodes = leaves
215            .into_iter()
216            .enumerate()
217            .chunks(ARITY)
218            .into_iter()
219            .map(|chunk| {
220                let children = chunk
221                    .map(|(pos, elem)| {
222                        let pos = pos as u64;
223                        Ok(Arc::new(MerkleNode::Leaf {
224                            value: H::digest_leaf(&pos, elem.borrow())?,
225                            pos,
226                            elem: elem.borrow().clone(),
227                        }))
228                    })
229                    .pad_using(ARITY, |_| Ok(Arc::new(MerkleNode::Empty)))
230                    .collect::<Result<Vec<_>, MerkleTreeError>>()?;
231                Ok(Arc::new(MerkleNode::<E, u64, T>::Branch {
232                    value: digest_branch::<E, H, u64, T>(&children)?,
233                    children,
234                }))
235            })
236            .collect::<Result<Vec<_>, MerkleTreeError>>()?;
237        for _ in 1..height {
238            cur_nodes = cur_nodes
239                .into_iter()
240                .chunks(ARITY)
241                .into_iter()
242                .map(|chunk| {
243                    let children: Vec<_> = chunk
244                        .pad_using(ARITY, |_| Arc::new(MerkleNode::<E, u64, T>::Empty))
245                        .collect();
246                    Ok(Arc::new(MerkleNode::<E, u64, T>::Branch {
247                        value: digest_branch::<E, H, u64, T>(&children)?,
248                        children,
249                    }))
250                })
251                .collect::<Result<Vec<_>, MerkleTreeError>>()?;
252        }
253        Ok((cur_nodes[0].clone(), height, num_leaves))
254    }
255}

257#[allow(clippy::type_complexity)]
258pub(crate) fn build_light_weight_tree_internal<E, H, const ARITY: usize, T>(
259    height: Option<usize>,
260    elems: impl IntoIterator<Item = impl Borrow<E>>,
261) -> Result<(Arc<MerkleNode<E, u64, T>>, usize, u64), MerkleTreeError>
262where
263    E: Element,
264    H: DigestAlgorithm<E, u64, T>,
265    T: NodeValue,
266{
267    let leaves: Vec<_> = elems.into_iter().collect();
268    let num_leaves = leaves.len() as u64;
269    let height = height.unwrap_or_else(|| {
270        let mut height = 0usize;
271        let mut capacity = 1;
272        while capacity < num_leaves {
273            height += 1;
274            capacity *= ARITY as u64;
275        }
276        height
277    });
278    let capacity = num_traits::checked_pow(ARITY as u64, height).ok_or_else(|| {
279        MerkleTreeError::ParametersError("Merkle tree size too large.".to_string())
280    })?;
281
282    if num_leaves > capacity {
283        Err(MerkleTreeError::ExceedCapacity)
284    } else if num_leaves == 0 {
285        Ok((Arc::new(MerkleNode::<E, u64, T>::Empty), height, 0))
286    } else if height == 0usize {
287        Ok((
288            Arc::new(MerkleNode::Leaf {
289                value: H::digest_leaf(&0, leaves[0].borrow())?,
290                pos: 0,
291                elem: leaves[0].borrow().clone(),
292            }),
293            height,
294            1,
295        ))
296    } else {
297        let mut cur_nodes = leaves
298            .into_iter()
299            .enumerate()
300            .chunks(ARITY)
301            .into_iter()
302            .map(|chunk| {
303                let children = chunk
304                    .map(|(pos, elem)| {
305                        let pos = pos as u64;
306                        Ok(if pos < num_leaves - 1 {
307                            Arc::new(MerkleNode::ForgottenSubtree {
308                                value: H::digest_leaf(&pos, elem.borrow())?,
309                            })
310                        } else {
311                            Arc::new(MerkleNode::Leaf {
312                                value: H::digest_leaf(&pos, elem.borrow())?,
313                                pos,
314                                elem: elem.borrow().clone(),
315                            })
316                        })
317                    })
318                    .pad_using(ARITY, |_| Ok(Arc::new(MerkleNode::Empty)))
319                    .collect::<Result<Vec<_>, MerkleTreeError>>()?;
320                Ok(Arc::new(MerkleNode::<E, u64, T>::Branch {
321                    value: digest_branch::<E, H, u64, T>(&children)?,
322                    children,
323                }))
324            })
325            .collect::<Result<Vec<_>, MerkleTreeError>>()?;
326        for i in 1..cur_nodes.len() - 1 {
327            cur_nodes[i] = Arc::new(MerkleNode::ForgottenSubtree {
328                value: cur_nodes[i].value(),
329            })
330        }
331        for _ in 1..height {
332            cur_nodes = cur_nodes
333                .into_iter()
334                .chunks(ARITY)
335                .into_iter()
336                .map(|chunk| {
337                    let children = chunk
338                        .pad_using(ARITY, |_| Arc::new(MerkleNode::<E, u64, T>::Empty))
339                        .collect::<Vec<_>>();
340                    Ok(Arc::new(MerkleNode::<E, u64, T>::Branch {
341                        value: digest_branch::<E, H, u64, T>(&children)?,
342                        children,
343                    }))
344                })
345                .collect::<Result<Vec<_>, MerkleTreeError>>()?;
346            for i in 1..cur_nodes.len() - 1 {
347                cur_nodes[i] = Arc::new(MerkleNode::ForgottenSubtree {
348                    value: cur_nodes[i].value(),
349                })
350            }
351        }
352        Ok((cur_nodes[0].clone(), height, num_leaves))
353    }
354}

356pub(crate) fn digest_branch<E, H, I, T>(
357    data: &[Arc<MerkleNode<E, I, T>>],
358) -> Result<T, MerkleTreeError>
359where
360    E: Element,
361    H: DigestAlgorithm<E, I, T>,
362    I: Index,
363    T: NodeValue,

376    /// Forget a leaf from the merkle tree. Internal branch merkle node will
377    /// also be forgotten if all its leaves are forgotten.
378    /// WARN(#495): this method breaks non-membership proofs.
379    #[allow(clippy::type_complexity)]
380    pub(crate) fn forget_internal(
381        &self,
382        height: usize,
383        traversal_path: &[usize],
384    ) -> (
385        Arc<Self>,
386        LookupResult<E, MerkleTreeProof<T>, MerkleTreeProof<T>>,
387    ) {
388        match self {
389            MerkleNode::Empty => (
390                Arc::new(self.clone()),
391                LookupResult::NotFound(MerkleTreeProof(vec![])),
392            ),
393            MerkleNode::Branch { value, children } => {
394                let mut children = children.clone();
395                let (new_child, result) = children[traversal_path[height - 1]]
396                    .forget_internal(height - 1, traversal_path);
397                match result {
398                    LookupResult::Ok(elem, mut membership_proof) => {
399                        membership_proof.0.push(
400                            children
401                                .iter()
402                                .enumerate()
403                                .filter(|(id, _)| *id != traversal_path[height - 1])
404                                .map(|(_, child)| child.value())
405                                .collect::<Vec<_>>(),
406                        );
407                        children[traversal_path[height - 1]] = new_child;
408                        if children.iter().all(|child| {
409                            matches!(
410                                **child,
411                                MerkleNode::Empty | MerkleNode::ForgottenSubtree { .. }
412                            )
413                        }) {
414                            (
415                                Arc::new(MerkleNode::ForgottenSubtree { value: *value }),
416                                LookupResult::Ok(elem, membership_proof),
417                            )
418                        } else {
419                            (
420                                Arc::new(MerkleNode::Branch {
421                                    value: *value,
422                                    children,
423                                }),
424                                LookupResult::Ok(elem, membership_proof),
425                            )
426                        }
427                    },
428                    LookupResult::NotInMemory => {
429                        (Arc::new(self.clone()), LookupResult::NotInMemory)
430                    },
431                    LookupResult::NotFound(mut non_membership_proof) => {
432                        non_membership_proof.0.push(
433                            children
434                                .iter()
435                                .enumerate()
436                                .filter(|(id, _)| *id != traversal_path[height - 1])
437                                .map(|(_, child)| child.value())
438                                .collect::<Vec<_>>(),
439                        );
440                        (
441                            Arc::new(self.clone()),
442                            LookupResult::NotFound(non_membership_proof),
443                        )
444                    },
445                }
446            },
447            MerkleNode::Leaf {
448                value,
449                pos: _,
450                elem,
451            } => (
452                Arc::new(MerkleNode::ForgottenSubtree { value: *value }),
453                LookupResult::Ok(elem.clone(), MerkleTreeProof(vec![])),
454            ),
455            _ => (Arc::new(self.clone()), LookupResult::NotInMemory),
456        }
457    }

459    /// Re-insert a forgotten leaf to the Merkle tree.
460    /// It also fails if the Merkle proof is invalid.
461    pub(crate) fn remember_internal<H, const ARITY: usize>(
462        &self,
463        height: usize,
464        traversal_path: &[usize],
465        pos: &I,
466        element: Option<&E>,
467        proof: &[Vec<T>],
468    ) -> Result<Arc<Self>, MerkleTreeError>
469    where
470        H: DigestAlgorithm<E, I, T>,
471    {
472        match self {
473            MerkleNode::Empty => Ok(Arc::new(self.clone())),
474            MerkleNode::Leaf { .. } => {
475                if height != 0 {
476                    // Reach a leaf before it should
477                    Err(MerkleTreeError::InconsistentStructureError(
478                        "Malformed Merkle tree or proof".to_string(),
479                    ))
480                } else {
481                    Ok(Arc::new(self.clone()))
482                }
483            },
484            MerkleNode::Branch { value, children } => {
485                if height == 0 {
486                    // Reach a branch when there should be a leaf
487                    Err(MerkleTreeError::InconsistentStructureError(
488                        "Malformed merkle tree".to_string(),
489                    ))
490                } else {
491                    let branch = traversal_path[height - 1];
492                    let mut children = children.clone();
493                    children[branch] = children[branch].remember_internal::<H, ARITY>(
494                        height - 1,
495                        traversal_path,
496                        pos,
497                        element,
498                        proof,
499                    )?;
500                    Ok(Arc::new(MerkleNode::Branch {
501                        value: *value,
502                        children,
503                    }))
504                }
505            },
506            MerkleNode::ForgottenSubtree { value } => Ok(Arc::new(if height == 0 {
507                if let Some(element) = element {
508                    MerkleNode::Leaf {
509                        value: H::digest_leaf(pos, element)?,
510                        pos: pos.clone(),
511                        elem: element.clone(),
512                    }
513                } else {
514                    MerkleNode::Empty
515                }
516            } else {
517                let branch = traversal_path[height - 1];
518                let mut values = proof[height - 1].clone();
519                values.insert(branch, *value);
520                let mut children = values
521                    .iter()
522                    .map(|&value| Arc::new(MerkleNode::ForgottenSubtree { value }))
523                    .collect::<Vec<_>>();
524                children[branch] = children[branch].remember_internal::<H, ARITY>(
525                    height - 1,
526                    traversal_path,
527                    pos,
528                    element,
529                    proof,
530                )?;
531                values[branch] = children[branch].value();
532                MerkleNode::Branch {
533                    value: H::digest(&values)?,
534                    children,
535                }
536            })),
537        }
538    }

540    /// Query the given index at the current Merkle node. Return the element
541    /// with a membership proof if presence, otherwise return a non-membership
542    /// proof.
543    #[allow(clippy::type_complexity)]
544    pub(crate) fn lookup_internal(
545        &self,
546        height: usize,
547        traversal_path: &[usize],
548    ) -> LookupResult<&E, MerkleTreeProof<T>, MerkleTreeProof<T>> {
549        match self {
550            MerkleNode::Empty => LookupResult::NotFound(MerkleTreeProof(vec![vec![]; height])),
551            MerkleNode::Branch { value: _, children } => {
552                match children[traversal_path[height - 1]]
553                    .lookup_internal(height - 1, traversal_path)
554                {
555                    LookupResult::Ok(elem, mut membership_proof) => {
556                        membership_proof.0.push(
557                            children
558                                .iter()
559                                .enumerate()
560                                .filter(|(id, _)| *id != traversal_path[height - 1])
561                                .map(|(_, child)| child.value())
562                                .collect::<Vec<_>>(),
563                        );
564                        LookupResult::Ok(elem, membership_proof)
565                    },
566                    LookupResult::NotInMemory => LookupResult::NotInMemory,
567                    LookupResult::NotFound(mut non_membership_proof) => {
568                        non_membership_proof.0.push(
569                            children
570                                .iter()
571                                .enumerate()
572                                .filter(|(id, _)| *id != traversal_path[height - 1])
573                                .map(|(_, child)| child.value())
574                                .collect::<Vec<_>>(),
575                        );
576                        LookupResult::NotFound(non_membership_proof)
577                    },
578                }
579            },
580            MerkleNode::Leaf {
581                elem,
582                value: _,
583                pos: _,
584            } => LookupResult::Ok(elem, MerkleTreeProof(vec![])),
585            _ => LookupResult::NotInMemory,
586        }
587    }

589    /// Update the element at the given index.
590    /// * `returns` - `Err()` if any error happens internally. `Ok(delta,
591    ///   result)`, `delta` represents the changes to the overall number of
592    ///   leaves of the tree, `result` contains the original lookup information
593    ///   at the given location.
594    #[allow(clippy::type_complexity)]
595    pub(crate) fn update_with_internal<H, const ARITY: usize, F>(
596        &self,
597        height: usize,
598        pos: impl Borrow<I>,
599        traversal_path: &[usize],
600        f: F,
601    ) -> Result<(Arc<Self>, i64, LookupResult<E, (), ()>), MerkleTreeError>
602    where
603        H: DigestAlgorithm<E, I, T>,
604        F: FnOnce(Option<&E>) -> Option<E>,
605    {
606        let pos = pos.borrow();
607        match self {
608            MerkleNode::Leaf {
609                elem: node_elem,
610                value: _,
611                pos,
612            } => {
613                let result = LookupResult::Ok(node_elem.clone(), ());
614                match f(Some(node_elem)) {
615                    Some(elem) => Ok((
616                        Arc::new(MerkleNode::Leaf {
617                            value: H::digest_leaf(pos, &elem)?,
618                            pos: pos.clone(),
619                            elem,
620                        }),
621                        0i64,
622                        result,
623                    )),
624                    None => Ok((Arc::new(MerkleNode::Empty), -1i64, result)),
625                }
626            },
627            MerkleNode::Branch { value, children } => {
628                let branch = traversal_path[height - 1];
629                let result = children[branch].update_with_internal::<H, ARITY, _>(
630                    height - 1,
631                    pos,
632                    traversal_path,
633                    f,
634                )?;
635                let mut children = children.clone();
636                children[branch] = result.0;
637                if matches!(*children[branch], MerkleNode::ForgottenSubtree { .. }) {
638                    // If the branch containing the update was forgotten by
639                    // user, the update failed and nothing was changed, so we
640                    // can short-circuit without recomputing this node's value.
641                    Ok((
642                        Arc::new(MerkleNode::Branch {
643                            value: *value,
644                            children,
645                        }),
646                        result.1,
647                        result.2,
648                    ))
649                } else if children
650                    .iter()
651                    .all(|child| matches!(**child, MerkleNode::Empty))
652                {
653                    Ok((Arc::new(MerkleNode::Empty), result.1, result.2))
654                } else {
655                    // Otherwise, an entry has been updated and the value of one of our children has
656                    // changed, so we must recompute our own value.
657                    // *value = digest_branch::<E, H, I, T>(&children)?;
658                    Ok((
659                        Arc::new(MerkleNode::Branch {
660                            value: digest_branch::<E, H, I, T>(&children)?,
661                            children,
662                        }),
663                        result.1,
664                        result.2,
665                    ))
666                }
667            },
668            MerkleNode::Empty => {
669                if height == 0 {
670                    if let Some(elem) = f(None) {
671                        Ok((
672                            Arc::new(MerkleNode::Leaf {
673                                value: H::digest_leaf(pos, &elem)?,
674                                pos: pos.clone(),
675                                elem,
676                            }),
677                            1i64,
678                            LookupResult::NotFound(()),
679                        ))
680                    } else {
681                        Ok((
682                            Arc::new(MerkleNode::Empty),
683                            0i64,
684                            LookupResult::NotFound(()),
685                        ))
686                    }
687                } else {
688                    let branch = traversal_path[height - 1];
689                    let mut children: Vec<_> = (0..ARITY).map(|_| Arc::new(Self::Empty)).collect();
690                    // Inserting new leave here, shortcutting
691                    let result = children[branch].update_with_internal::<H, ARITY, _>(
692                        height - 1,
693                        pos,
694                        traversal_path,
695                        f,
696                    )?;
697                    children[branch] = result.0;
698                    if matches!(*children[branch], MerkleNode::Empty) {
699                        // No update performed.
700                        Ok((Arc::new(MerkleNode::Empty), 0i64, result.2))
701                    } else {
702                        Ok((
703                            Arc::new(MerkleNode::Branch {
704                                value: digest_branch::<E, H, I, T>(&children)?,
705                                children,
706                            }),
707                            result.1,
708                            result.2,
709                        ))
710                    }
711                }
712            },
713            MerkleNode::ForgottenSubtree { .. } => Err(MerkleTreeError::ForgottenLeaf),
714        }
715    }
716}
717
718impl<E, T> MerkleNode<E, u64, T>
719where
720    E: Element,
721    T: NodeValue,
722{
723    /// Batch insertion for the given Merkle node.
724    pub(crate) fn extend_internal<H, const ARITY: usize>(
725        &self,
726        height: usize,
727        pos: &u64,
728        traversal_path: &[usize],
729        at_frontier: bool,
730        data: &mut Peekable<impl Iterator<Item = impl Borrow<E>>>,
731    ) -> Result<(Arc<Self>, u64), MerkleTreeError>
732    where
733        H: DigestAlgorithm<E, u64, T>,
734    {
735        if data.peek().is_none() {
736            return Ok((Arc::new(self.clone()), 0));
737        }
738        let mut cur_pos = *pos;
739        match self {
740            MerkleNode::Branch { value: _, children } => {
741                let mut cnt = 0u64;
742                let mut frontier = if at_frontier {
743                    traversal_path[height - 1]
744                } else {
745                    0
746                };
747                let cap = ARITY;
748                let mut children = children.clone();
749                while data.peek().is_some() && frontier < cap {
750                    let (new_child, increment) = children[frontier].extend_internal::<H, ARITY>(
751                        height - 1,
752                        &cur_pos,
753                        traversal_path,
754                        at_frontier && frontier == traversal_path[height - 1],
755                        data,
756                    )?;
757                    children[frontier] = new_child;
758                    cnt += increment;
759                    cur_pos += increment;
760                    frontier += 1;
761                }
762                let value = digest_branch::<E, H, u64, T>(&children)?;
763                Ok((Arc::new(Self::Branch { value, children }), cnt))
764            },
765            MerkleNode::Empty => {
766                if height == 0 {
767                    let elem = data.next().unwrap();
768                    let elem = elem.borrow();
769                    Ok((
770                        Arc::new(MerkleNode::Leaf {
771                            value: H::digest_leaf(pos, elem)?,
772                            pos: *pos,
773                            elem: elem.clone(),
774                        }),
775                        1,
776                    ))
777                } else {
778                    let mut cnt = 0u64;
779                    let mut frontier = if at_frontier {
780                        traversal_path[height - 1]
781                    } else {
782                        0
783                    };
784                    let cap = ARITY;
785                    let mut children: Vec<_> = (0..cap).map(|_| Arc::new(Self::Empty)).collect();
786                    while data.peek().is_some() && frontier < cap {
787                        let (new_child, increment) = children[frontier]
788                            .extend_internal::<H, ARITY>(
789                                height - 1,
790                                &cur_pos,
791                                traversal_path,
792                                at_frontier && frontier == traversal_path[height - 1],
793                                data,
794                            )?;
795                        children[frontier] = new_child;
796                        cnt += increment;
797                        cur_pos += increment;
798                        frontier += 1;
799                    }
800                    Ok((
801                        Arc::new(MerkleNode::Branch {
802                            value: digest_branch::<E, H, u64, T>(&children)?,
803                            children,
804                        }),
805                        cnt,
806                    ))
807                }
808            },
809            MerkleNode::Leaf { .. } => Err(MerkleTreeError::ExistingLeaf),
810            MerkleNode::ForgottenSubtree { .. } => Err(MerkleTreeError::ForgottenLeaf),
811        }
812    }
813
814    /// Similar to [`extend_internal`], but this function will automatically
815    /// forget every leaf except for the Merkle tree frontier.
816    pub(crate) fn extend_and_forget_internal<H, const ARITY: usize>(
817        &self,
818        height: usize,
819        pos: &u64,
820        traversal_path: &[usize],
821        at_frontier: bool,
822        data: &mut Peekable<impl Iterator<Item = impl Borrow<E>>>,
823    ) -> Result<(Arc<Self>, u64), MerkleTreeError>
824    where
825        H: DigestAlgorithm<E, u64, T>,
826    {
827        if data.peek().is_none() {
828            return Ok((Arc::new(self.clone()), 0));
829        }
830        let mut cur_pos = *pos;
831        match self {
832            MerkleNode::Branch { value: _, children } => {
833                let mut cnt = 0u64;
834                let mut frontier = if at_frontier {
835                    traversal_path[height - 1]
836                } else {
837                    0
838                };
839                let cap = ARITY;
840                let mut children = children.clone();
841                while data.peek().is_some() && frontier < cap {
842                    if frontier > 0 && !children[frontier - 1].is_forgotten() {
843                        children[frontier - 1] =
844                            Arc::new(MerkleNode::<E, u64, T>::ForgottenSubtree {
845                                value: children[frontier - 1].value(),
846                            });
847                    }
848                    let (new_child, increment) = children[frontier]
849                        .extend_and_forget_internal::<H, ARITY>(
850                            height - 1,
851                            &cur_pos,
852                            traversal_path,
853                            at_frontier && frontier == traversal_path[height - 1],
854                            data,
855                        )?;
856                    children[frontier] = new_child;
857                    cnt += increment;
858                    cur_pos += increment;
859                    frontier += 1;
860                }
861                let value = digest_branch::<E, H, u64, T>(&children)?;
862                Ok((Arc::new(Self::Branch { value, children }), cnt))
863            },
864            MerkleNode::Empty => {
865                if height == 0 {
866                    let elem = data.next().unwrap();
867                    let elem = elem.borrow();
868                    Ok((
869                        Arc::new(MerkleNode::Leaf {
870                            value: H::digest_leaf(pos, elem)?,
871                            pos: *pos,
872                            elem: elem.clone(),
873                        }),
874                        1,
875                    ))
876                } else {
877                    let mut cnt = 0u64;
878                    let mut frontier = if at_frontier {
879                        traversal_path[height - 1]
880                    } else {
881                        0
882                    };
883                    let cap = ARITY;
884                    let mut children: Vec<_> = (0..cap).map(|_| Arc::new(Self::Empty)).collect();
885                    while data.peek().is_some() && frontier < cap {
886                        if frontier > 0 && !children[frontier - 1].is_forgotten() {
887                            children[frontier - 1] =
888                                Arc::new(MerkleNode::<E, u64, T>::ForgottenSubtree {
889                                    value: children[frontier - 1].value(),
890                                });
891                        }
892                        let (new_child, increment) = children[frontier]
893                            .extend_and_forget_internal::<H, ARITY>(
894                                height - 1,
895                                &cur_pos,
896                                traversal_path,
897                                at_frontier && frontier == traversal_path[height - 1],
898                                data,
899                            )?;
900                        children[frontier] = new_child;
901                        cnt += increment;
902                        cur_pos += increment;
903                        frontier += 1;
904                    }
905                    Ok((
906                        Arc::new(MerkleNode::Branch {
907                            value: digest_branch::<E, H, u64, T>(&children)?,
908                            children,
909                        }),
910                        cnt,
911                    ))
912                }
913            },
914            MerkleNode::Leaf { .. } => Err(MerkleTreeError::ExistingLeaf),
915            MerkleNode::ForgottenSubtree { .. } => Err(MerkleTreeError::ForgottenLeaf),
916        }
917    }
918}

920/// Iterator type for a merkle tree
921pub struct MerkleTreeIter<'a, E: Element, I: Index, T: NodeValue> {
922    stack: Vec<&'a MerkleNode<E, I, T>>,
923}
924
925impl<'a, E: Element, I: Index, T: NodeValue> MerkleTreeIter<'a, E, I, T> {
926    /// Initialize an iterator
927    pub fn new(root: &'a MerkleNode<E, I, T>) -> Self {
928        Self { stack: vec![root] }
929    }
930}
931
932impl<'a, E, I, T> Iterator for MerkleTreeIter<'a, E, I, T>
933where
934    E: Element,
935    I: Index,
936    T: NodeValue,
937{
938    type Item = (&'a I, &'a E);
939
940    fn next(&mut self) -> Option<Self::Item> {
941        while let Some(node) = self.stack.pop() {
942            match node {
943                MerkleNode::Branch { value: _, children } => {
944                    children
945                        .iter()
946                        .rev()
947                        .filter(|child| {
948                            matches!(
949                                ***child,
950                                MerkleNode::Branch { .. } | MerkleNode::Leaf { .. }
951                            )
952                        })
953                        .for_each(|child| self.stack.push(child));
954                },
955                MerkleNode::Leaf {
956                    value: _,
957                    pos,
958                    elem,
959                } => {
960                    return Some((pos, elem));
961                },
962                _ => {},
963            }
964        }
965        None
966    }
967}
